## 3. 创建主题切换组件 (`src/components/ThemeSwitcher.vue`)

```vue
<template>
  <div class="dropdown dropdown-end">
    <div tabindex="0" role="button" class="btn btn-ghost btn-circle">
      <Icon 
        :icon="SunnyIcon" 
        theme="outline" 
        size="20" 
        :fill="currentTheme === 'dark' ? '#fbbf24' : '#374151'"
      />
    </div>
    <ul tabindex="0" class="dropdown-content z-[1] menu p-2 shadow-2xl bg-base-100 rounded-box w-52 gap-1">
      <li v-for="theme in themes" :key="theme.id">
        <button 
          @click="setTheme(theme.id)"
          class="flex items-center gap-3"
          :class="{ 'active': currentTheme === theme.id }"
        >
          <div 
            class="w-6 h-6 rounded-full" 
            :style="{ backgroundColor: theme.color }"
          ></div>
          <span>{{ theme.name }}</span>
          <Icon 
            v-if="currentTheme === theme.id"
            :icon="CheckOneIcon"
            theme="outline"
            size="16"
            class="ml-auto text-primary"
          />
        </button>
      </li>
    </ul>
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'
import { Sunny, CheckOne } from '@icon-park/vue-next'

// IconPark 图标
const SunnyIcon = Sunny
const CheckOneIcon = CheckOne

// 主题列表
const themes = [
  { id: 'light', name: '简约白', color: '#3b82f6' },
  { id: 'dark', name: '高级黑', color: '#6366f1' },
  { id: 'pink', name: '少女粉', color: '#ec489a' },
  { id: 'green', name: '清新绿', color: '#10b981' },
  { id: 'blue', name: '海洋蓝', color: '#3b82f6' },
]

const currentTheme = ref('light')

// 设置主题
const setTheme = (theme) => {
  currentTheme.value = theme
  document.documentElement.setAttribute('data-theme', theme)
  localStorage.setItem('theme', theme)
}

// 初始化主题
onMounted(() => {
  const savedTheme = localStorage.getItem('theme') || 'light'
  setTheme(savedTheme)
})
</script>

<style scoped>
.active {
  @apply bg-primary/10 text-primary;
}
</style>
```

## 4. 创建图标展示组件 (`src/components/IconDemo.vue`)

```vue
<template>
  <div class="grid grid-cols-4 sm:grid-cols-6 md:grid-cols-8 lg:grid-cols-12 gap-4 p-4">
    <div 
      v-for="icon in icons" 
      :key="icon.name"
      class="flex flex-col items-center gap-2 p-3 rounded-lg hover:bg-base-200 transition-colors cursor-pointer"
      @click="copyIconName(icon.name)"
    >
      <component 
        :is="icon.component" 
        theme="outline" 
        size="28"
        :fill="currentColor"
        :strokeWidth="3"
      />
      <span class="text-xs text-center">{{ icon.name }}</span>
    </div>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue'
import * as Icons from '@icon-park/vue-next'

// 常用图标列表
const iconList = [
  'Home', 'User', 'Setting', 'Search', 'Bell', 'Message', 'Heart', 'Star',
  'Camera', 'Music', 'Video', 'Game', 'Book', 'Coffee', 'Car', 'Plane',
  'Sunny', 'Moon', 'Cloud', 'Rain', 'Snow', 'Wind', 'Fire', 'Lightning',
  'Add', 'Minus', 'Close', 'Check', 'Warning', 'Info', 'Help', 'Error',
  'Edit', 'Delete', 'Copy', 'Share', 'Download', 'Upload', 'Refresh', 'Lock',
  'Phone', 'Email', 'Location', 'Calendar', 'Clock', 'Chart', 'Wallet', 'Shop'
]

// 动态导入图标组件
const icons = iconList.map(name => ({
  name,
  component: Icons[name + 'One'] || Icons[name] || Icons[name + 'Filled']
})).filter(icon => icon.component)

// 当前主题颜色
const currentColor = computed(() => {
  const theme = document.documentElement.getAttribute('data-theme') || 'light'
  const colors = {
    light: '#3b82f6',
    dark: '#6366f1',
    pink: '#ec489a',
    green: '#10b981',
    blue: '#3b82f6'
  }
  return colors[theme] || '#3b82f6'
})

// 复制图标名称
const copyIconName = (name) => {
  navigator.clipboard.writeText(name)
  // 可以添加提示
  console.log(`已复制图标名: ${name}`)
}
</script>
```

## 7. 在 main.js 中全局注册 IconPark（可选）

```javascript
import { createApp } from 'vue'
import { createPinia } from 'pinia'
import router from './router'
import App from './App.vue'
import './style.css'

// 可选：全局注册 IconPark 图标库
import * as Icons from '@icon-park/vue-next'

const app = createApp(App)

// 全局注册所有图标（按需使用）
for (const [key, component] of Object.entries(Icons)) {
  app.component(key, component)
}

app.use(createPinia())
app.use(router)
app.mount('#app')
```
